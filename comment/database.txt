データベース命令について
CREATE TABLE IF NOT EXISTS visitors; = (存在しなければ)visitorsテーブルを作成

CREATE TABLE IF NOT EXISTS visitors (id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT); = idが主キー(自動入力)、nameが値

INSERT INTO visitors(name) VALUES('a'); = visitorsテーブルに新しい値を入れる(name="a")

SELECT name FROM visitors; //visitorsテーブルからname列を取得
